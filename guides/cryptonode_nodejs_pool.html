Setup Cryptonote-NodeJS-Pool

Step 1:

Update system 

sudo apt update

sudo apt upgrade

Step 2:

Install build requirements 

sudo apt-get install git redis python2 build-essential libssl-dev libboost-all-dev libsodium-dev curl

Step 3:

Install nvm 

curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.3/install.sh | bash

source ~/.bashrc

Step 4:

Install NodeJs 11.0.0 

nvm install v11.0.0

Step 5:

Set npm to use python 2.7 

npm config set python /usr/bin/python2.7

Step 6:

Clone cryptonote-nodejs-pool repository 

git clone https://github.com/kevacoin-team/cryptonote-nodejs-pool.git pool

Step 7:

Navigate to pool 

cd pool/

Step 8:

Run npm update 

npm update

Step 9:

Copy kevacoin.json to config.json 

cp ./config_examples/kevacoin.json config.json

Step 10:

Update config.json 

nano config.json

Step 11:

Run pool 

node init.js
